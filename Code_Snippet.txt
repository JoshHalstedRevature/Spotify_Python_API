          })

          authConnection = httplib.HTTPSConnection('api.spotify.com')

          authConnection.request(
               'GET',
               url = 'https://accounts.spotify.com/authorize?' + params
          )

          response = authConnection.getresponse()

          print(response.status, response.reason)
